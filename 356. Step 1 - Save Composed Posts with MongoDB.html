рЃюрЃљрЃЉрЃўрЃ»рЃў 1 - рЃерЃћрЃўрЃюрЃљрЃ«рЃћрЃЌ рЃерЃћрЃЊрЃњрЃћрЃюрЃўрЃџрЃў рЃърЃЮрЃАрЃбрЃћрЃЉрЃў MongoDB-рЃерЃў

рЃЏрЃўрЃќрЃљрЃюрЃў: рЃерЃћрЃю рЃБрЃюрЃЊрЃљ рЃерЃћрЃњрЃћрЃФрЃџрЃЮрЃА, рЃарЃЮрЃЏ рЃерЃћрЃ«рЃЋрЃўрЃЊрЃћ compose рЃњрЃЋрЃћрЃарЃЊрЃќрЃћ рЃЊрЃљ рЃЊрЃљрЃљрЃЏрЃљрЃбрЃЮ рЃљрЃ«рЃљрЃџрЃў рЃърЃЮрЃАрЃбрЃў, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃерЃћрЃўрЃюрЃљрЃ«рЃћрЃЉрЃљ рЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃЌрЃљ рЃЉрЃљрЃќрЃљрЃерЃў рЃарЃЮрЃњрЃЮрЃарЃф рЃЊрЃЮрЃЎрЃБрЃЏрЃћрЃюрЃбрЃў.



рЃўрЃЏрЃўрЃАрЃЌрЃЋрЃўрЃА рЃарЃЮрЃЏ рЃЊрЃљрЃарЃгрЃЏрЃБрЃюрЃЊрЃћ, рЃарЃЮрЃЏ рЃърЃЮрЃАрЃбрЃў рЃюрЃљрЃЏрЃЊрЃЋрЃўрЃџрЃљрЃЊ рЃерЃћрЃЊрЃў mongo shell-рЃерЃўрЃљ, рЃўрЃърЃЮрЃЋрЃћ рЃДрЃЋрЃћрЃџрЃљ рЃърЃЮрЃАрЃбрЃў.



рЃЕрЃљрЃАрЃЦрЃарЃЮрЃџрЃћ рЃЏрЃўрЃюрЃўрЃерЃюрЃћрЃЉрЃўрЃАрЃЌрЃЋрЃўрЃА ­ЪЉЄ




























  • рЃерЃћрЃю рЃБрЃюрЃЊрЃљ рЃЊрЃљрЃљрЃўрЃюрЃАрЃбрЃљрЃџрЃўрЃарЃЮ mongoose рЃЊрЃљ require(рЃерЃћрЃЏрЃЮрЃўрЃбрЃљрЃюрЃЮ) рЃўрЃА рЃерЃћрЃюрЃА app.js

  • рЃерЃћрЃю рЃЊрЃљрЃњрЃГрЃўрЃарЃЊрЃћрЃЉрЃљ рЃарЃЮрЃЏ рЃЊрЃљрЃБрЃЎрЃЮрЃюрЃћрЃЦрЃбрЃЊрЃћ рЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃЌрЃљ рЃЉрЃљрЃќрЃљрЃА рЃАрЃљрЃ«рЃћрЃџрЃљрЃЊ: blogDB


mongoose.connect("mongodb://localhost:27017/blogDB", {useNewUrlParser: true});


  • рЃерЃћрЃю рЃБрЃюрЃЊрЃљ рЃерЃћрЃЦрЃЏрЃюрЃљ рЃљрЃ«рЃљрЃџрЃў postSchema рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃерЃћрЃўрЃфрЃљрЃЋрЃА title(рЃАрЃљрЃЌрЃљрЃБрЃарЃА) рЃЊрЃљ content(рЃЎрЃЮрЃюрЃбрЃћрЃюрЃбрЃА).


const postSchema = {

 title: String,

 content: String

};


  • рЃерЃћрЃю рЃБрЃюрЃЊрЃљ рЃерЃћрЃЦрЃЏрЃюрЃљ рЃљрЃ«рЃљрЃџрЃў mongoose model, schema-рЃА рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃўрЃЌ рЃарЃљрЃЌрЃљ рЃњрЃљрЃљрЃЎрЃћрЃЌрЃЮ рЃърЃЮрЃАрЃбрЃћрЃЉрЃўрЃА рЃЎрЃЮрЃџрЃћрЃЦрЃфрЃўрЃўрЃА рЃЊрЃћрЃцрЃўрЃюрЃўрЃфрЃўрЃљ.


const Post = mongoose.model("Post", postSchema);



  • app.post() рЃЏрЃћрЃЌрЃЮрЃЊрЃерЃў, рЃерЃћрЃюрЃў /compose рЃарЃЮрЃБрЃбрЃўрЃАрЃЌрЃЋрЃўрЃА, рЃерЃћрЃю рЃБрЃюрЃЊрЃљ рЃерЃћрЃЦрЃЏрЃюрЃљ рЃљрЃ«рЃљрЃџрЃў рЃърЃЮрЃАрЃбрЃўрЃА рЃЊрЃЮрЃЎрЃБрЃЏрЃћрЃюрЃбрЃў, mongoose model-рЃўрЃА рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃўрЃЌ.


 const post = new Post ({

   title: req.body.postTitle,

   content: req.body.postBody

 });


  • рЃерЃћрЃю рЃБрЃюрЃЊрЃљ рЃерЃћрЃўрЃюрЃљрЃ«рЃЮ рЃћрЃА рЃЊрЃЮрЃЎрЃБрЃЏрЃћрЃюрЃбрЃў рЃерЃћрЃюрЃА рЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃЌрЃљ рЃЉрЃљрЃќрЃљрЃерЃў, рЃърЃЮрЃАрЃбрЃћрЃЉрЃўрЃА рЃЏрЃљрЃАрЃўрЃЋрЃерЃў рЃерЃћрЃюрЃљрЃ«рЃЋрЃўрЃА рЃюрЃљрЃфрЃЋрЃџрЃљрЃЊ.


post.save()
